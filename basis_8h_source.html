70 Basis combine(const Basis &basis, int order, double accScale = 1e-6,
71 double accBps = 1e-6) const {
72 assert(std::abs(m_scale - basis.getScale()) < accScale &&
73 "Only basis with same scale can be combined.");
74 assert(order >= std::max(m_order, basis.order()) &&
75 "New basis cannot subceed maximum of this and other bases order.");
76
77 // create this and other bases knots considering target order
78 Eigen::ArrayXd knotsThis{toKnots(getBreakpoints(accBps), order)};
79 Eigen::ArrayXd knotsOther{toKnots(basis.getBreakpoints(accBps), order)};
80
81 // combined knots cannot exceed size of this and other basis knots
82 Eigen::ArrayXd knotsComb(knotsThis.size() + knotsOther.size());
83
84 // iterate over both knot arrays
85 // compare knots and place smaller one in "knotsComb"
86 // auto knotComb {knotsComb.begin()};
87 auto knotThis{knotsThis.begin()};
88 auto knotOther{knotsOther.begin()};
89 int numKnotsComb{};
90 for (auto &knotComb : knotsComb) {
91 // test if terminal knot of this and other basis are reached
92 bool atThisEnd{knotThis == (knotsThis.end())};
93 bool atOtherEnd{knotOther == (knotsOther.end())};
94
95 // terminate since all knots are processed
96 if (atThisEnd && atOtherEnd)
97 break;
98
99 // assign this knot if smaller or other end is reached
100 if (*knotThis < *knotOther - accScale && !atThisEnd || atOtherEnd)
101 knotComb = *(knotThis++);
102 // assign other knot if smaller or other end is reached
103 else if (*knotOther < *knotThis - accScale && !atOtherEnd || atThisEnd)
104 knotComb = *(knotOther++);
105 // asign this and other knot, which are equal
106 else {
107 knotComb = *knotOther;
108
109 // count other knots since not fully processed
110 if (!atOtherEnd)
111 ++knotOther;
112
113 // count this knots since not fully processed
114 if (!atThisEnd)
115 ++knotThis;
116 }
117
118 ++numKnotsComb;
119 }
120
121 return {knotsComb(Eigen::seqN(0, numKnotsComb)), order};
122 }

150 Basis orderIncrease(int change = 1) const {
151 assert(change >= 0 && "Order change must be positive.");
152
153 // base case: no order increase, create new instance of current basis
154 if (change == 0)
155 return Basis{*this};
156
157 // create new knots where the first and last knots are completed according
158 // to order change
159 Eigen::ArrayXd knotsNew(knots().size() + 2 * change);
160 knotsNew << Eigen::ArrayXd::Zero(change) + knots()(0), knots(),
161 Eigen::ArrayXd::Zero(change) + *(knots().end() - 1);
162
163 // create basis of higher order
164 return {knotsNew, order() + change};

200 Eigen::MatrixXd derivative(Basis &basis, int orderDer = 1) const {
201 assert(orderDer >= 0 && "Derivative order must be positive.");
202
203 if (orderDer == 0) {
204 basis = *this;
205 return Eigen::MatrixXd::Identity(dim(), dim());
206 }
207
208 // determine transformation matrix
209 Eigen::MatrixXd transform(Eigen::MatrixXd::Zero(dim() - 1, dim()));
210 for (int cRow{}; cRow < transform.rows(); ++cRow) {
211 transform(cRow, cRow) = (order() - 1) /
212 (knots()(cRow + 1) - knots()(order() + cRow)) /
213 m_scale;
214 transform(cRow, cRow + 1) = -transform(cRow, cRow);
215 }
216
217 // provide basis derivative basis with decreased order
218 Basis basisDeriv{orderDecrease()};
219
220 // base case order 1 derivative
221 if (orderDer == 1) {
222 basis = basisDeriv;
223 return transform;
224 }
225
226 // recursion higher order derivative
227 return basisDeriv.derivative(basis, orderDer - 1) * transform;
228 };

241 Eigen::MatrixXd derivative(Basis &basis, const Eigen::MatrixXd &values,
242 int orderDer = 1) const {
243 assert(orderDer >= 0 && "Derivative order must be positive.");
244
245 if (orderDer == 0) {
246 basis = *this;
247 return values;
248 }
249
250 // provide basis derivative basis with decreased order
251 Basis basisDeriv{orderDecrease()};
252
253 // values transformed to derivative valuesNew = o * (values_i+1 - values_i)
254 // / (k_i+o - k_i+1)
255 Eigen::MatrixXd valuesNew(basisDeriv.dim(), values.cols());
256 for (int idx{}; idx < valuesNew.rows(); ++idx)
257 valuesNew(idx, Eigen::all) =
258 (order() - 1) *
259 (values(idx + 1, Eigen::all) - values(idx, Eigen::all)) /
260 (knots()(idx + order()) - knots()(idx + 1)) / m_scale;
261
262 // base case order 1 derivative
263 if (orderDer == 1) {
264 basis = basisDeriv;
265 return valuesNew;
266 }
267
268 // recursion higher order derivative
269 return basisDeriv.derivative(basis, valuesNew, orderDer - 1);
270 };

284 Eigen::MatrixXd integral(Basis &basis, int orderInt = 1) const {
285 assert(orderInt >= 0 && "Integral order must be positive.");
286
287 if (orderInt == 0) {
288 basis = *this;
289 return Eigen::MatrixXd::Identity(dim(), dim());
290 }
291
292 // initialize transformation matrix with zeros
293 Eigen::MatrixXd transform(Eigen::MatrixXd::Zero(dim() + 1, dim()));
294
295 // fill transformation matrix
296 int cCol{};
297 for (auto col : transform.colwise()) {
298 for (int cRow{cCol + 1}; cRow < transform.rows(); ++cRow) {
299 col(cRow) =
300 (knots()(order() + cCol) - knots()(cCol)) / order() * m_scale;
301 }
302 ++cCol;
303 }
304
305 // provide basis integral basis with increased order
306 Basis basisDeriv{orderIncrease()};
307
308 // base case order 1 integral
309 if (orderInt == 1) {
310 basis = basisDeriv;
311 return transform;
312 }
313
314 // recursion higher order integral
315 return basisDeriv.integral(basis, orderInt - 1) * transform;
316 }

329 Eigen::MatrixXd integral(Basis &basis, const Eigen::MatrixXd &values,
330 int orderInt = 1) const {
331 assert(orderInt >= 0 && "Integral order must be positive.");
332
333 if (orderInt == 0) {
334 basis = *this;
335 return values;
336 }
337
338 // provide basis integral basis with decreased order
339 Basis basisInt{orderIncrease()};
340
341 // values transformed to integral valuesNew_i+1 = values_i * (k_i+o -
342 // k_i) / o + valuesNew_i
343 Eigen::MatrixXd valuesNew{
344 Eigen::MatrixXd::Zero(basisInt.dim(), values.cols())};
345 for (int idx{}; idx < valuesNew.rows() - 1; ++idx)
346 valuesNew(idx + 1, Eigen::all) =
347 values(idx, Eigen::all) * (knots()(idx + order()) - knots()(idx)) /
348 order() * m_scale +
349 valuesNew(idx, Eigen::all);
350
351 // base case order 1 integral
352 if (orderInt == 1) {
353 basis = basisInt;
354 return valuesNew;
355 }
356
357 // recursion higher order integral
358 return basisInt.integral(basis, valuesNew, orderInt - 1);

499 Eigen::MatrixXd operator()(const Eigen::ArrayXd &points, double accBps = 1e-6,
500 double accSegment = 1e-6) const {
501 // stores evaluation of truncated powers at given points
502 Eigen::MatrixXd basisValues{Eigen::MatrixXd::Zero(points.size(), dim())};
503
504 // evaluate trunctated powers for each point
505 int cPoint{};
506 for (double point : points) {
507 // each VectorXd stores values of bases of increasing order
508 std::vector<Eigen::VectorXd> basesValues(m_order);
509
510 // evaluate basis of order 1 which is eiter 1.0 or 0.0
511 basesValues[0].resize(m_knots.size() - 1);
512 for (int cKnot{}; cKnot < m_knots.size() - 1; ++cKnot)
513 basesValues[0](cKnot) =
514 inKnotSeg(m_knots(cKnot), m_knots(cKnot + 1), point, accSegment)
515 ? 1.0
516 : 0.0;
517
518 // evaluate bases of order > 1 in ascending order
519 for (int cOrder{2}; cOrder <= m_order; ++cOrder) {
520 // get basis values of next higher order as weighted sum of neighboring
521 // basis values
522 for (int cKnot{}; cKnot < m_knots.size() - cOrder; ++cKnot) {
523 // determine basis weight based on current knot
524 const double denumCurr{m_knots(cKnot + cOrder - 1) - m_knots(cKnot)};
525 const double weightCurr{std::abs(denumCurr) > accBps
526 ? (point - m_knots(cKnot)) / denumCurr
527 : 0.0};
528
529 // determine basis weight based on next knot
530 const double denumNext{m_knots(cKnot + cOrder) - m_knots(cKnot + 1)};
531 const double weightNext{std::abs(denumNext) > accBps
532 ? (m_knots(cKnot + cOrder) - point) /
533 denumNext
534 : 0.0};
535
536 // basis value of higher order
537 basesValues[cOrder - 1].resize(m_knots.size() - cOrder);
538 basesValues[cOrder - 1](cKnot) =
539 weightCurr * basesValues[cOrder - 2](cKnot) +
540 weightNext * basesValues[cOrder - 2](cKnot + 1);
541 }
542 }
543
544 // store maximum order basis values for current point
545 basisValues(cPoint++, Eigen::seqN(0, dim())) =
546 basesValues[m_order - 1](Eigen::seqN(0, dim()));
547 }
548
549 return basisValues;
550 }

685 static Eigen::ArrayXd toKnots(const Eigen::ArrayXd &breakpoints,
686 const Eigen::ArrayXi &continuities, int order) {
687 // instantiate counters for breakpoint knot multiplicity
688 Eigen::ArrayXi mults{order - continuities};
689 // instantiate knot result array
690 Eigen::ArrayXd knots(breakpoints.size() * order - continuities.sum());
691
692 // assign knots according to breakpoints and their multiplicity
693 auto mult{mults.begin()};
694 auto bp{breakpoints.begin()};
695 for (double &knot : knots) {
696 knot = *bp;
697 --(*mult);
698 if (*mult == 0) {
699 ++mult;
700 ++bp;
701 }
702 }
703
704 return knots;
705 }

732 static std::pair<Eigen::ArrayXd, Eigen::ArrayXi>
733 toBreakpoints(const Eigen::ArrayXd &knots, int order,
734 double accuracy = 1e-6) {
735 // instantiate breakpoints, which cannot exceed the number of breakpoints
736 Eigen::ArrayXd breakpoints(knots.size());
737 // first breakpoint must equal first knot
738 breakpoints(0) = knots(0);
739
740 // instantiate breakpoint continuities
741 Eigen::ArrayXi continuities{Eigen::ArrayXi::Zero(knots.size()) + order};
742 // reduce first breakpoint continuity since it corresponds to first knot
743 --continuities(0);
744
745 // assign knots to breakpoints and reduce continuity per assigned knot
746 int idxBps{};
747 auto knot{knots.begin() + 1};
748 for (; knot != knots.end(); ++knot) {
749 // assign knot to breakpoint if in accuracy
750 if (*knot > breakpoints(idxBps) + accuracy)
751 breakpoints(++idxBps) = *knot;
752 --continuities(idxBps);
753 }
754
755 // return visited breakpoints and continuities
756 return {breakpoints(Eigen::seqN(0, idxBps + 1)),
757 continuities(Eigen::seqN(0, idxBps + 1))};
758 }

781 bool inKnotSeg(double knotL, double knotR, double point,
782 double accPoint = 1e-6, double accKnot = 1e-6) const {
783
784 const bool knotlEqsFirst{std::abs(knotL - m_knots(0)) <= accKnot};
785 const bool knotrEqsLast{std::abs(knotR - m_knots(m_knots.size() - 1)) <=
786 accKnot};
787
788 if (knotlEqsFirst && knotrEqsLast)
789 return point > knotL - accPoint && point <= knotR + accPoint;
790 else if (knotlEqsFirst)
791 return point >= knotL - accPoint && point <= knotR;
792 else if (knotrEqsLast)
793 return point > knotL && point <= knotR + accPoint;
794 return point > knotL && point <= knotR;
795 }

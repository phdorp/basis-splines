56 Eigen::MatrixXd fit(const std::vector<Eigen::MatrixXd> &observations,
57 const std::vector<Eigen::VectorXi> &derivOrders,
58 const Eigen::ArrayXd &points) const {
59
60 // store transformation matrices and spline bases
61 std::vector<Eigen::MatrixXd> transforms(m_basis->order() - 1);
62 std::vector<Basis> bases(m_basis->order() - 1);
63
64 // initialize transformations and bases with current
65 transforms[0] = Eigen::MatrixXd::Identity(m_basis->dim(), m_basis->dim());
66 bases[0] = *m_basis;
67
68 // create higher order derivatives based on previous one
69 int cOrder{};
70 for (int cOrder{1}; cOrder < m_basis->order() - 1; ++cOrder)
71 transforms[cOrder] = bases[cOrder - 1].derivative(bases[cOrder], 1);
72
73 // evaluate basis functions at "points" and transform according to given
74 // derivartive
75 Eigen::MatrixXd basisValues(m_basis->dim(), m_basis->dim());
76 int cObs{};
77 int cRow{};
78 for (const Eigen::MatrixXd &observation : observations) {
79 int cValue{};
80 for (auto row : observation.rowwise()) {
81 basisValues(cRow++, Eigen::all) =
82 bases[derivOrders[cObs](cValue)]({{points(cObs)}}) *
83 transforms[derivOrders[cObs](cValue)];
84 ++cValue;
85 }
86 ++cObs;
87 }
88
89 // arrange observation in array
90 Eigen::MatrixXd splineValues(m_basis->dim(), observations[0].cols());
91 int cElem{};
92 for (const auto &observation : observations)
93 for (auto row : observation.rowwise())
94 splineValues(cElem++, Eigen::all) = row;
95
96 // solve for spline coefficients
97 return DecompositionType{basisValues}.solve(splineValues);
98 }

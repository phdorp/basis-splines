147 template <typename Interp = Interpolate>
148 Spline add(const Spline &other, double accScale = 1e-6,
149 double accBps = 1e-6) const {
150 // combine this and other basis to new basis
151 // new basis order is maximum of this and other basis order
152 const std::shared_ptr<Basis> newBasis{std::make_shared<Basis>(
153 m_basis->combine(*other.basis().get(),

155 accScale, accBps))};

157 // determine coefficients by interpolating sum of this and other spline
158 return {newBasis, Interp{newBasis}.fit([&](const Eigen::ArrayXd &points) {
159 return Eigen::MatrixXd{(*this)(points) + other(points)};

335 Eigen::VectorXd interpolateCoefficients(double knotInsert, int dim) const {
336 Eigen::VectorXd coeffsNew(m_coefficients.rows() + 1);
337 const Eigen::VectorXd coeffs{m_coefficients(Eigen::all, dim)};
338 const Eigen::ArrayXd knots{m_basis->knots()};
339 const Eigen::Index order{m_basis->order()};
340
341 Eigen::Index knotIdx{};
342
343 // case 1: copy coefficients
344 while (knotInsert >= knots(knotIdx + order - 1)) {
345 coeffsNew(knotIdx) = coeffs(knotIdx);
346 ++knotIdx;
347 }
348
349 // case 2: interpolate coefficients
350 while (knots(knotIdx) < knotInsert &&
351 knotInsert < knots(knotIdx + order - 1)) {
352 double weight{(knotInsert - knots(knotIdx)) /
353 (knots(knotIdx + order - 1) - knots(knotIdx))};
354 coeffsNew(knotIdx) =
355 (1 - weight) * coeffs(knotIdx - 1) + weight * coeffs(knotIdx);
356 ++knotIdx;
357 }
358
359 // case 3: shift coefficients
360 for (; knotIdx < coeffsNew.size(); ++knotIdx)
361 coeffsNew(knotIdx) = coeffs(knotIdx - 1);
362
363 return coeffsNew;
364 }

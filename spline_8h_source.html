148 Spline add(const Spline &other, double accScale = 1e-6,
149 double accBps = 1e-6) const {
150 // combine this and other basis to new basis
151 // new basis order is maximum of this and other basis order
152 const std::shared_ptr<Basis> newBasis{std::make_shared<Basis>(
153 m_basis->combine(*other.basis().get(),
154 std::max(m_basis->order(), other.basis()->order()),
155 accScale, accBps))};
156
157 // determine coefficients by interpolating sum of this and other spline
158 return {newBasis, Interp{newBasis}.fit([&](const Eigen::ArrayXd &points) {
159 return Eigen::MatrixXd{(*this)(points) + other(points)};
160 })};

387 Eigen::VectorXd interpolateCoefficients(double knotInsert, int dim) const {
388 Eigen::VectorXd coeffsNew(m_coefficients.rows() + 1);
389 const Eigen::VectorXd coeffs{m_coefficients(Eigen::all, dim)};
390 const Eigen::ArrayXd knots{m_basis->knots()};
391 const Eigen::Index order{m_basis->order()};
392
393 Eigen::Index knotIdx{};
394
395 // case 1: copy coefficients
396 while (knotInsert >= knots(knotIdx + order - 1)) {
397 coeffsNew(knotIdx) = coeffs(knotIdx);
398 ++knotIdx;
399 }
400
401 // case 2: interpolate coefficients
402 while (knots(knotIdx) < knotInsert &&
403 knotInsert < knots(knotIdx + order - 1)) {
404 double weight{(knotInsert - knots(knotIdx)) /
405 (knots(knotIdx + order - 1) - knots(knotIdx))};
406 coeffsNew(knotIdx) =
407 (1 - weight) * coeffs(knotIdx - 1) + weight * coeffs(knotIdx);
408 ++knotIdx;
409 }
410
411 // case 3: shift coefficients
412 for (; knotIdx < coeffsNew.size(); ++knotIdx)
413 coeffsNew(knotIdx) = coeffs(knotIdx - 1);
414
415 return coeffsNew;
416 }

439 double estimateRoot(int rootIdx, int dim, int maxIter, double accAbs) const {
440 Spline inserted{*this};
441 Eigen::VectorXd coeffs{inserted.getCoefficients()(Eigen::all, dim)};
442 double rootGuess{inserted.basis()->greville(rootIdx)};
443
444 for (int iter = 0; iter < maxIter && !isRoot(rootGuess, dim, accAbs);
445 iter++) {
446 double leftCoeff{coeffs(rootIdx)};
447 double leftGrev{inserted.basis()->greville(rootIdx)};
448 bool leftAlmostZero{abs(leftCoeff) <= accAbs};
449
450 // left greville is root
451 if (leftAlmostZero)
452 rootGuess = leftGrev;
453
454 // guess root between left and right greville
455 else {
456 double rightCoeff{coeffs(rootIdx + 1)};
457 double rightGrev{inserted.basis()->greville(rootIdx + 1)};
458 bool rightAlmostZero{abs(rightCoeff) <= accAbs};
459
460 double param{-leftCoeff / (rightCoeff - leftCoeff)};
461 rootGuess = (1 - param) * leftGrev + param * rightGrev;
462
463 inserted = inserted.insertKnots(Eigen::ArrayXd{{rootGuess}});
464 coeffs = inserted.getCoefficients()(Eigen::all, dim);
465
466 // after insertion, check if zero moved between new greville and former
467 // right greville
468 if (coeffs(rootIdx + 1) * coeffs(rootIdx + 2) <= 0)
469 rootIdx += 1;
470 }
471 }
472
473 return rootGuess;
474 };

489 std::pair<Eigen::ArrayXi, Eigen::ArrayXd> getRootIdcs(int dim,
490 double absTol) const {
491 const Eigen::VectorXd coeffs{m_coefficients(Eigen::all, dim)};
492 const int maxRoots = coeffs.size();
493
494 Eigen::ArrayXi rootIdcs = Eigen::ArrayXi::Constant(maxRoots, -1);
495 Eigen::ArrayXd roots(maxRoots);
496 int cntRoots{};
497
498 for (Eigen::Index cntCoeff{}; cntCoeff < maxRoots; ++cntCoeff) {
499 double coeff{coeffs(cntCoeff)};
500 bool coeffAlmostZero{abs(coeff) <= absTol};
501
502 // root candidate at coefficient
503 if (coeffAlmostZero) {
504 double greville = m_basis->greville(cntCoeff);
505
506 if (isRoot(greville, dim, absTol))
507 roots(cntRoots++) = greville;
508 }
509
510 // root between coefficients
511 else if (cntCoeff < maxRoots - 1) {
512 double nextCoeff{coeffs(cntCoeff + 1)};
513 bool nextNotZero{abs(nextCoeff) > absTol};
514 bool containsRoot{coeff * nextCoeff <= 0};
515
516 if (nextNotZero && containsRoot)
517 rootIdcs(cntRoots++) = cntCoeff;
518 }
519 }
520
521 return {rootIdcs.head(cntRoots), roots.head(cntRoots)};
522 }
